

Download Teleport
wget https://github.com/gravitational/teleport/releases/download/v1.2.0/teleport-v1.2.0-linux-amd64-bin.tar.gz

tar -xvf teleport-v1.2.0-linux-amd64-bin.tar.gz
cd teleport
sudo make install

whoami

# Start the auth server
sudo teleport start --roles=auth

# On the same machine add users
tctl users add vagrant vagrant
